expression → equality ;
equality   → comparison ( ( "!=" | "==" ) comparison )* ;
comparison → term ( ( ">" | ">=" | "<" | "<=" ) term )* ;
term       → factor ( ( "-" | "+" ) factor )* ;
factor     → unary ( ( "/" | "*" ) unary )* ;
unary      → ( "!" | "-" ) unary | power ;
pow      → primary ( "^" pow )* ;
primary    → NUMBER | STRING | "true" | "false" | "nil" | "(" expression ")" ;
